```sql
-- What widgets do users move most often?
SELECT widget_type, COUNT(*) as move_count
FROM layout_change_events
WHERE change_type = 'widget_moved'
GROUP BY widget_type
ORDER BY move_count DESC;

-- Time-of-day patterns
SELECT
  (context->>'hour_of_day')::int as hour,
  COUNT(*) as customization_count
FROM layout_change_events
GROUP BY hour
ORDER BY hour;

-- Riley's Monday morning pattern
SELECT
  widget_type,
  new_position->>'x' as x,
  new_position->>'y' as y,
  COUNT(*) as frequency
FROM layout_change_events
WHERE user_id = 'riley-uuid'
  AND (context->>'day_of_week')::int = 1
  AND (context->>'hour_of_day')::int BETWEEN 6 AND 10
GROUP BY widget_type, x, y
ORDER BY frequency DESC;
```

```typescript
async function detectUniversityLMS(email: string): Promise<LMSConfig | null> {
  const domain = email.split('@')[1];

  // Check campus_configs for known mappings
  const campus = await db.campus_configs.findFirst({
    where: { email_domains: { contains: domain } }
  });

  if (campus?.canvas_instance_url) {
    return {
      type: 'canvas',
      instanceUrl: campus.canvas_instance_url,
      features: ['live_events', 'oauth', 'calendar_sync']
    };
  }

  return null;
}
```

```typescript
async function generateSuggestions(userId: string): Promise<LayoutSuggestion[]> {
  const patterns = await detectUserPatterns(userId);
  const currentContext = getCurrentContext();

  return patterns
    .filter(p => p.matchesContext(currentContext))
    .map(p => p.toSuggestion())
    .filter(s => s.confidence > 0.7);
}
```

## Scaling Considerations

### Data Volume

**Layout Events**:
- ~100 bytes per event
- ~10 events per user per day
- ~1KB per user per day
- 10,000 users = 10MB/day = 300MB/month

**Insight Interactions**:
- ~50 bytes per interaction
- ~20 interactions per user per day
- ~1KB per user per day

**Storage Strategy**:
- PostgreSQL for recent events (90 days)
- Archive older events to S3 (cost optimization)
- Aggregated patterns stay in PostgreSQL

### Performance

**Dashboard BFF**:
- Continue using ETag caching
- Insights computed in background, cached
- Real-time updates via Ably for urgent changes only

**Event Logging**:
- Client-side buffer (batch events)
- Debounced server sync (every 5 seconds)
- Background job for pattern detection (daily)

**Pattern Detection**:
- Run nightly batch job
- Cache patterns per user
- Invalidate on significant new events

### Privacy & Trust

**Core Principles**:
1. Explain the data, not the surveillance
2. Insights should help, not judge
3. User controls everything
4. Progressive trust-building
5. Attribution is mandatory

**Data Retention**:
- Layout events: 90 days hot, then archived
- Insight interactions: 30 days
- Patterns/preferences: Until user deletes

**User Controls**:
- Delete all learning data
- Export personal data
- Granular opt-out per insight type
- Disable all AI features

## Key Insights Summary

1. **Visual polish builds trust** - Users need to see quality before they trust AI
2. **Attribution removes fear** - Every insight needs a "why"
3. **Customization generates data** - Let users personalize, learn from choices
4. **Progressive disclosure** - Glanceable â†’ Expandable â†’ On-Demand
5. **Necessary frictions** - Some insights need opt-in gates (grades, behavioral)
6. **Never scary** - Help, don't judge; suggest, don't demand

**The Vision**: A dashboard that looks premium, provides genuinely useful AI insights with clear attribution, and learns from every interaction to become more personalized over time - all while keeping the user in control.
