## Segmentation Logic

```javascript
// Smart template selection based on user segment
function selectEmailTemplate(user: User, campaign: Campaign): Template {
  const segment = getUserSegment(user);
  const context = getUserContext(user);
  
  // Personality-based selection
  if (user.preferences.aiPersonality === 'motivational') {
    return getMotivationalVariant(campaign);
  }
  
  if (user.preferences.aiPersonality === 'straightforward') {
    return getMinimalVariant(campaign);
  }
  
  // Engagement-based selection
  if (segment.engagement === 'high') {
    return getDetailedVariant(campaign);
  }
  
  if (segment.engagement === 'low') {
    return getSimpleVariant(campaign);
  }
  
  // Context-based selection
  if (context.isExamWeek) {
    return getExamFocusedVariant(campaign);
  }
  
  if (context.isBreak) {
    return getLightVariant(campaign);
  }
  
  return getDefaultTemplate(campaign);
}
```
