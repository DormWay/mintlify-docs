async function generateStudySessionRecommendations(dayPlan: DayPlan, canvasData: CanvasData) {
  // Find assignments due this week
  const upcomingAssignments = canvasData.assignments.filter(a =>
    isWithinDays(a.due_at, 7)
  );

  // Calculate study time needed
  const studyTimeNeeded = upcomingAssignments.reduce((total, a) =>
    total + estimateStudyTime(a), 0
  );

  // Find optimal study slots from DayPlan energy curve
  const optimalSlots = dayPlan.flow.energyCurve
    .filter(slot => slot.energy > 0.7)  // High energy periods
    .filter(slot => !dayPlan.events.some(e => overlaps(e, slot)));

  // Schedule study sessions
  return optimalSlots.map(slot => ({
    type: 'study_session',
    start_time: slot.time,
    duration: Math.min(90, studyTimeNeeded),  // Max 90min blocks
    assignment_id: upcomingAssignments[0].id,
    reason: `Optimal study time based on your energy pattern`
  }));
}
```

**2. Smart Priority Scoring** (1 day)

// Add to academic page backend (services/api-router/src/routes/academic-routes.ts)
function calculateAcademicPriority(assignment: Assignment, context: StudentContext): number {
  let score = 0;

  // Time urgency (0-40 points)
  const hoursUntilDue = getHoursUntil(assignment.due_at);
  if (assignment.missing) score += 50;        // CRITICAL
  else if (hoursUntilDue < 12) score += 40;   // Due today
  else if (hoursUntilDue < 48) score += 30;   // Due tomorrow
  else if (hoursUntilDue < 168) score += 20;  // Due this week

  // Weight/importance (0-25 points)
  if (assignment.points_possible >= 100) score += 25;
  else if (assignment.points_possible >= 50) score += 20;
  else score += 10;

  // Performance context (0-20 points)
  const courseGrade = context.grades[assignment.course_id];
  if (courseGrade < 70) score += 20;  // Struggling in course
  else if (courseGrade < 80) score += 10;

  // Historical patterns (0-15 points)
  const pastPerformance = context.assignmentHistory[assignment.type];
  if (pastPerformance && pastPerformance.avgGrade < courseGrade) {
    score += 15;  // This type of assignment is harder for student
  }

  return score;
}
```
